def factorial(n: int) -> int:
    """–í—ã—á–∏—Å–ª–µ–Ω–∏–µ —Ñ–∞–∫—Ç–æ—Ä–∏–∞–ª–∞ —á–∏—Å–ª–∞ n —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–æ."""
    if n == 0 or n == 1: # –ë–∞–∑–æ–≤—ã–µ —Å–ª—É—á–∞–∏
        return 1
    return n * factorial(n - 1)

# –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(n)
# –ì–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏: O(h)


def fibonacci(n: int) -> int:
    """–†–µ–∫—É—Ä—Å–∏–≤–Ω–æ–µ –≤—ã—á–∏—Å–ª–µ–Ω–∏–µ n-–≥–æ —á–∏—Å–ª–∞ –§–∏–±–æ–Ω–∞—á—á–∏."""
    if n <= 1: # –ë–∞–∑–æ–≤—ã–µ —Å–ª—É—á–∞–∏ 
        return n
    return fibonacci(n - 1) + fibonacci(n - 2) # –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(2^n)

# –í—Ä–µ–º–µ–Ω–Ω–∞—è —Å–ª–æ–∂–Ω–æ—Å—Ç—å: O(2^n)
# –ì–ª—É–±–∏–Ω–∞ —Ä–µ–∫—É—Ä—Å–∏–∏: O(h)


def fast_power(a: float, n: int) -> float:
    """–ë—ã—Å—Ç—Ä–æ–µ –≤–æ–∑–≤–µ–¥–µ–Ω–∏–µ —á–∏—Å–ª–∞ a –≤ —Å—Ç–µ–ø–µ–Ω—å n (—á–µ—Ä–µ–∑ –¥–µ–ª–µ–Ω–∏–µ —Å—Ç–µ–ø–µ–Ω–∏ –Ω–∞ 2)."""
    if n == 0: # –ë–∞–∑–æ–≤—ã–µ —Å–ª—É—á–∞–∏ 
        return 1
    if n % 2 == 0:
        half = fast_power(a, n // 2) # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –≤—ã–∑–æ–≤
        return half * half
    else: # n –Ω–µ—á–µ—Ç–Ω–æ–µ
        return a * fast_power(a, n - 1) # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –≤—ã–∑–æ–≤

import os

def binary_search(arr, target, left=0, right=None):
    """–†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –±–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫."""
    if right is None: # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –ø—Ä–∞–≤–æ–π –≥—Ä–∞–Ω–∏—Ü—ã O(1)
        right = len(arr) - 1
    if left > right: # –ü—Ä–æ–≤–µ—Ä–∫–∞ –≥—Ä–∞–Ω–∏—Ü O(1)
        return -1
    mid = (left + right) // 2 # –í—ã—á–∏—Å–ª–µ–Ω–∏–µ —Å—Ä–µ–¥–Ω–µ–≥–æ –∏–Ω–¥–µ–∫—Å–∞ O(1)
    if arr[mid] == target: # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ —Ä–∞–≤–µ–Ω—Å—Ç–≤–æ O(1)
        return mid # –≠–ª–µ–º–µ–Ω—Ç –Ω–∞–π–¥–µ–Ω O(1)
    elif arr[mid] > target: # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –≤—ã–∑–æ–≤ –¥–ª—è –ª–µ–≤–æ–π –ø–æ–ª–æ–≤–∏–Ω—ã O(log n)
        return binary_search(arr, target, left, mid - 1) 
    else: # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –≤—ã–∑–æ–≤ –¥–ª—è –ø—Ä–∞–≤–æ–π –ø–æ–ª–æ–≤–∏–Ω—ã O(log n)
        return binary_search(arr, target, mid + 1, right)

def walk_directory(path, indent=0, current_depth=1):
    """–†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –æ–±—Ö–æ–¥ —Ñ–∞–π–ª–æ–≤–æ–π —Å–∏—Å—Ç–µ–º—ã —Å –≤—ã–≤–æ–¥–æ–º –¥–µ—Ä–µ–≤–∞ –∫–∞—Ç–∞–ª–æ–≥–æ–≤ –∏ —Ñ–∞–π–ª–æ–≤."""
    if not os.path.exists(path):
        print("–ü—É—Ç—å –Ω–µ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç")
        return 0
    max_found_depth = current_depth
    for item in sorted(os.listdir(path)):
        full_path = os.path.join(path, item)
        print(" " * indent + f"- {item} - {current_depth}")
        if os.path.isdir(full_path):
            sub_depth = walk_directory(full_path, indent + 2, current_depth + 1)
            max_found_depth = max(max_found_depth, sub_depth)
    return max_found_depth

def hanoi(n, source="A", target="C", auxiliary="B"):
    """–†–µ—à–µ–Ω–∏–µ –∑–∞–¥–∞—á–∏ –•–∞–Ω–æ–π—Å–∫–∏—Ö –±–∞—à–µ–Ω –¥–ª—è n –¥–∏—Å–∫–æ–≤."""
    if n == 1:
        print(f"–ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å –¥–∏—Å–∫ 1 —Å {source} –Ω–∞ {target}")
        return
    hanoi(n - 1, source, auxiliary, target)
    print(f"–ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å –¥–∏—Å–∫ {n} —Å {source} –Ω–∞ {target}")
    hanoi(n - 1, auxiliary, target, source)

def fibonacci_memo(n: int, memo=None) -> int:
    if memo is None: # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —Å–ª–æ–≤–∞—Ä—è –º–µ–º–æ–∏–∑–∞—Ü–∏–∏ –ø—Ä–∏ –ø–µ—Ä–≤–æ–º –≤—ã–∑–æ–≤–µ O(1)
        memo = {} # O(1)

    if n == 0: # –ë–∞–∑–æ–≤—ã–µ —Å–ª—É—á–∞–∏
        return 0 # O(1)
    if n == 1: # O(1) 
        return 1 # O(1)
    
    if n in memo: # –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω–µ –≤—ã—á–∏—Å–ª—è–ª–∏ –ª–∏ –º—ã —É–∂–µ —ç—Ç–æ –∑–Ω–∞—á–µ–Ω–∏–µ
        return memo[n] # O(1)

    memo[n] = fibonacci_memo(n - 1, memo) + fibonacci_memo(n - 2, memo) # –†–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π —à–∞–≥ —Å —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞ O(n)
    return memo[n] # O(1)

def max_depth_walk(path, current_depth=0):
    if not os.path.exists(path):
        return current_depth
    max_depth = current_depth
    for item in os.listdir(path):
        full_path = os.path.join(path, item)
        if os.path.isdir(full_path):
            max_depth = max(max_depth, max_depth_walk(full_path, current_depth + 1))
    return max_depth
